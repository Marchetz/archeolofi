# login
from base64 import b64encode
def make_basic_auth(user, psw):
    return "Basic " + b64encode("{0}:{1}".format(user, psw))


r = requests.get(
    url + "login/",
    headers={'Authorization': make_basic_auth(u["name"], u["psw"])}
